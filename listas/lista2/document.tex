 \paragraph{Exercício 1} Seja $S$ um vetor aleatório, e $X$ um conjunto de variáveis aleatórias. Mostre a lei da variância total, i.e:
 
 $$\mathbb{V}[S] = \mathbb{E}[\mathbb{V}[S|X]] + \mathbb{V}[\mathbb{E}[S|X]]\, .$$

 \paragraph{Exercício 2} Sejam $A$ e $B$ duas matrizes simétricas positiva definidas de dimensão $n$. Mostre que, se $A-B$ é positiva semidefinida, então $B^{-1} - A^{-1}$ é positiva semidefinida.
 
 \paragraph{Exercício 3} Considere o seguinte modelo linear para uma amostra de $n$ observações:
 
 $$\boldsymbol{y} = \boldsymbol{X}_1 \beta_1 + \boldsymbol{X}_2 \beta_2 + \boldsymbol{\epsilon}\,,$$
 onde $\mathbb{E}[\boldsymbol{\epsilon}|\boldsymbol{X}_1,\boldsymbol{X}_2]=0$, $\mathbb{V}[\boldsymbol{\epsilon}|\boldsymbol{X}]=\sigma^2\mathbb{I}_{n\times n}$, $\left[\boldsymbol{X}_1 \, \, \boldsymbol{X}_2\right]$ tem posto cheio, e $\boldsymbol{X}_1 = \left[ \boldsymbol{1}_{n\times 1} \, \, \boldsymbol{Z}\right]$, onde  $\boldsymbol{1}_{n\times 1}  $ é um vetor de uns e $\boldsymbol{Z}$ é \emph{independente} de $\boldsymbol{X}_2$. Suponha, ademais, por simplicidade, que $\boldsymbol{X}_2$ é normalizado de tal forma que $\mathbb{E}[\boldsymbol{X}_2] = 0_{n\times k_2}$ (do contrário, basta redefinir o intercepto para discutir o que segue).
 
 \begin{itemize}
 	\item[a] Mostre que o estimador de MQO de $\boldsymbol{y}$ em $\boldsymbol{X}_1$, denotado por $\tilde{b}_1$, é não viciado (condicionalmente a $\boldsymbol{X}_1$) para $\beta_1$.
 	\item[b] Mostre que a variância condicional de $\tilde{b}_1$ se escreve como:
 	
 	$$\mathbb{V}[\tilde{b}_1|\boldsymbol{X}_1] = 	(\boldsymbol{X}_1'\boldsymbol{X}_1)^{-1}\boldsymbol{X}_1'\mathbb{V}[\boldsymbol{X}_2\beta_2|\boldsymbol{X}_1]\boldsymbol{X}_1'(\boldsymbol{X}_1\boldsymbol{X}_1)^{-1} + \sigma^2 (\boldsymbol{X}_1'\boldsymbol{X}_1)^{-1}$$
 	
 	\item[c] Mostre que a variância do estimador de MQO de $\beta_1$ da regressão que inclui $\boldsymbol{X}_1$ e $\boldsymbol{X}_2$, denotado por $\hat{b}_1$, é não viciado para $\beta_1$ (condicionalmente a $\boldsymbol{X}_1$ e $\boldsymbol{X}_2$) e que sua variância condicional é dada por:
 	
 	$$\mathbb{V}[\hat{b}_1|\boldsymbol{X}_1,\boldsymbol{X}_2] = \sigma^2\left(\boldsymbol{X}_1'M_2 \boldsymbol{X}_1\right)^{-1}\, , $$
 	onde $M_2$ é a residualizadora de $\boldsymbol{X}_2$.
 	
 	\item[d] Mostre que $\left(\boldsymbol{X}_1'M_2 \boldsymbol{X}_1\right)^{-1} - (\boldsymbol{X}_1'\boldsymbol{X}_1)^{-1}$ é positiva semidefinida.
 	
 	\item[e] Mostre que $(\boldsymbol{X}_1'\boldsymbol{X}_1)^{-1}\boldsymbol{X}_1'\mathbb{V}[\boldsymbol{X}_2\beta_2]\boldsymbol{X}_1'(\boldsymbol{X}_1'\boldsymbol{X}_1)^{-1} $ é positiva semidefinida.
 	
 	\item[f] Usando a  lei da variância total e os dois resultados anteriores, mostre que a diferença entre as variâncias \emph{incondicionais} de $\tilde{b}_1$ e $\hat{b}_1$ se fatoram em dois termos, $A$ e $B$, em que $A$ é a esperança de uma matriz aleatória positiva semidefinida e $B$ a esparança de uma matriz negativa semidefinida. Proveja uma intuição para esse resultado. Se o interesse reside em $\beta_1$, quando fará sentido incluir $\boldsymbol{X}_2$ na regressão?
 	\item[g] Mostre que se $\beta_2 = 0$, então  $\mathbb{V}[\hat{\beta}_1] - \mathbb{V}[\tilde{\beta}_1] $ é positiva semidefinida. Esse resultado contradiz o teorema de Gauss-Markov? Por quê?
 \end{itemize}
 
 
 \paragraph{Exercício 4} Considere um modelo linear da forma:
 
 $$\boldsymbol{y} = \boldsymbol{X}\beta + \boldsymbol{\epsilon}$$
 onde $\boldsymbol{y}$ é um vetor aleatório $n\times 1$, $\boldsymbol{X} = \begin{bmatrix}
 	X_1' \\
 	X_2' \\
 	\vdots \\
 	X_n'
 \end{bmatrix}$ é uma matriz aleatória $n \times k$ e $\boldsymbol{\epsilon}$ é um vetor aleatório de dimensão $n \times 1$. No que segue, vamos supor que $\boldsymbol{X}$ tem posto $k$ (com probabilidade 1), e que a hipótese de exogeneidade $\mathbb{E}[\boldsymbol{\epsilon}|\boldsymbol{X}] = 0$ é satisfeita. Suponha, também, que a matriz de variância  (condicional) dos $\boldsymbol{\epsilon}$ seja diagonal (i.e. os erros são condicionalmente não correlacionados), embora potencialmente heterocedástica, i.e.:
 
 $$\mathbb{V}[\boldsymbol{\epsilon}|\boldsymbol{X}] = \begin{bmatrix}
 	\sigma^2_1(\boldsymbol{X}) & 0 & 0 & \ldots & 0 \\
 	0 & 	\sigma^2_2(\boldsymbol{X}) & 0 & \ldots & 0 \\
 	\vdots & \vdots &\ddots & \ldots & \vdots  \\
 	0 & 0 & 0 & \ldots & \sigma^2_n(\boldsymbol{X})
 \end{bmatrix}\, ,$$
 onde os $\sigma^2_i$ são funções desconhecidas de $\boldsymbol{X}$. 
 \begin{itemize}
 	\item[a] Mostre que  a matriz de variância do estimador de MQO de $\beta$, denotado por $\hat{b}$ toma a forma:
 	
 	$$\mathbb{V}[\hat{b}|\boldsymbol{X}] = (\boldsymbol{X}'\boldsymbol{X})^{-1}\left(\sum_{i=1}^n X_i X_i' \sigma_i^2(\boldsymbol{X})\right)(\boldsymbol{X}'\boldsymbol{X})^{-1}$$
 	
 	\item[b] Suponha que a matriz $\boldsymbol{X}$ é robusta a eliminações de uma observação da amostra, no sentido de que as matrizes $(n-1)\times k$ que consistem na remoção da $i$-ésima linha de $\boldsymbol{X}$ e que denotamos por $\boldsymbol{X}_{-i}$ têm posto $k$, para todo $i=1\ldots, n$. Considere o seguinte resíduo modificado:
 	
 	$$\tilde{e}_i = (Y_i - X_i'\hat{b}_{-i})\,,$$
 	onde $\hat{b}_{-i}$ é o estimador de MQO de $\beta$ na subamostra de $(n-1)$ observações que exclui $i$. Mostre que (com probabilidade 1):
 	$$\mathbb{E}[\tilde{e}_i^2|\boldsymbol{X}]\geq \sigma^2_i(\boldsymbol{X})\, .$$
 	
 	\item[c]  Mostre que o resíduo convencional, $\hat{e}_i = Y_i-X_i'\hat{b}$, não necessariamente satisfaz $\mathbb{E}[\hat{e}_i^2|\boldsymbol{X}]\geq \sigma^2_i(\boldsymbol{X})$ com probabilidade 1. Dê uma intuição para esse resultado.
 	\item[d]  Com base nos resultados anteriores, construa um estimador $\hat{V}$ para a variância $\mathbb{V}[\hat{b}|\boldsymbol{X}]$ (em média) conservador, no sentido de que (com probabilidade 1):
 	$$\mathbb{E}[\hat{V}|\boldsymbol{X}] - \mathbb{V}[\hat{b}|\boldsymbol{X}]\, ,$$
 	é positiva semidefinida. Argumente que o estimador \textit{heteroskedasticity-consistent} de White para a variância visto em aula não necessariamente satisfará essa propriedade.
 \end{itemize}

 \paragraph{Exercício 5} Considere variáveis aleatórias $Y$ e $D$ cuja distribuição conjunta representa a distribuição, em uma população de interesse, de um resultado de interesse $Y$, e uma causa binária ($D$) cujos efeitos causais desejamos aferir. Defina as variáveis aleatórias $Y(0)$ e $Y(1)$ correspondentes ao experimento hipotético de se amostrar um indivíduo da população e manipular (\textit{ceteris paribus}) a causa binária para $0$ ($Y(0)$) e $1$ ($Y(1)$), de modo que o resultado observado é dado por $Y = DY(1)+(1-D)Y(0)$. Vamos supor que $D$ é independente de $(Y(0),Y(1))$, de modo que não há associação sistemática, na população, entre o valor observado da causa binária e os resultados que seriam observados caso o tratamento fosse fixado externamente em $0$ ou em $1$. 
 
 \begin{itemize}
 	\item[a] Mostre que $Y$ pode ser escrito como:
 	$$Y = \alpha + \beta D + U \, ,$$
 	onde $\beta = \mathbb{E}[Y(1)-Y(0)]$ é o efeito esperado de se amostrar um indivíduo da população e avaliar o efeito da causa $D$ sobre ele; $\mathbb{E}[U] =0$ e $\operatorname{cov}(D,U) =0$. 
 	\item[b]   Mostre que o estimador de MQO do coeficiente associado a $D$ numa regressão de $Y$ num intercepto, com base numa amostra aleatória de $(D,Y)$, é consistente para $\beta$. O estimador é não viciado para $\beta$? Justifique.
 	\item[c] Mostre que, se não houver heterogeneidade no efeito do tratamento, i.e. $\mathbb{V}[Y(1)-Y(0)]=0$, então existirá $\sigma^2$ tal que a variável aleatória $U$ do modelo definido no item (a) satisfaz: $\mathbb{V}[U|D] = \sigma^2$ (com probabilidade 1).
 	\item[d] Com base no resultado do item anterior e numa amostra aleatória de $(D,Y)$, sugira um teste para a hipótese nula de que não há heterogeneidade no efeito do tratamento, contra a alternativa de que há.
 \end{itemize}
 
 
 
 
 \paragraph{Exercício 6} O arquivo \texttt{cps\_union\_data.csv} conté uma amostra \textbf{12.834 indivíduos na força de trabalho norte-americana}, extraída do suplemento anual de 2019 da \textit{Pesquisa de População Atual dos EUA (Current Population Survey)}. O conjunto de dados contém diversas outras variáveis que podem ser utilizadas como controles (verifique o dicionário do conjunto de dados, disponível em \texttt{dictionary.xlsx}).

Qual é a diferença estimada?   \paragraph{Exercício 7} Suponha que você esteja interessado em estudar o efeito de infraestrutura escolar no desempenho escolar médio dos aluno. Você possui acesso a uma amostra aleatória de municípios, para os quais você observa o desempenho escolar médio dos alunos de $C$ escolas, $c=1,\ldots, C$, em cada um dos $M$ munícipios,  $m=1,\ldots, M$, amostrados ($Y_{c,m}$), e uma medida de conectividade da escola à Internet, em \textit{megabytes por segundo} (mbps) ($X_{c,m}$). Você então postula o seguinte modelo causal linear para $Y_{c,m}$
  
  $$Y_{c,m} = \gamma X_{c,m} + \epsilon_{c,m}\,,\quad c=1, \ldots,  C, \, \, m=1,\ldots, M\, ,$$
 onde $\gamma$ é o efeito causal de se aumentar a conectivadade da escola em 1mbps sobre o desempenho médio dos alunos, e $\epsilon_{c,m}$ são as demais causas não observadas do desempenho escolar.
 
 \begin{itemize}
 	\item[a] Sob qual condição a inclinação do melhor preditor linear de $Y_{c,m}$ num intercepto e $X_{c,m}$ identifica $\gamma$? Você acredita que essa condição seria satisfeita na prática? Por quê?
 \end{itemize}
 \vspace{0.5 em}
 
 Em virtude da dificuldade de se acreditar na hipótese anteriormente postulada, um colega seu sugere que você explicitamente acomode a presença de causas não observadas no nível municipal que afetam desempenho municipal, considerando a decomposição:
 
 $$\epsilon_{c,m} = \psi_m + \nu_{c,m}\, , $$
 onde $\psi_m$ é um conjunto de causas comuns ao município, e $\nu_{c,m}$ são causas idiossincráticas às escolas. Seu colega lhe diz que é possível levar em conta  (``controlar'') as causas $\psi_m$ na estimação, mesmo que estas não sejam observadas, considerando uma regressão de $Y_{c,m}$ em $X_{c,m}$ e um conjunto de $M$ \textit{dummies} municipais, i.e. $D_{c,m}(l) =\mathbf{1}_{\{l\}}(m)$, $l=1\,\ldots, m$.
 
 \begin{itemize}
 	\item[b] Usando o teorema de Frisch-Waugh-Lovell, mostre que o estimador do coeficiente de $X_{c,m}$ sugerido pelo seu colega é algebricamente idêntico a se estimar $\gamma$ a partir de uma regressão (sem intercepto) de $Y_{c,m} - \bar{Y}_m$ em $X_{c,m} - \bar{X}_m$, onde $\bar{Z}_m = \frac{\sum_{i=1}^{C}}{C} Z_{i.m}$. Em outras palavras, o estimador sugeriod pelo seu colega é equivalente ao estimador de MQO do modelo transformado.
 	
 	$$Y_{c,m} - \bar{Y}_m = \gamma (X_{c,m}-\bar{X}_m) + \epsilon_{c,m}-\bar{\epsilon}_m\, .$$
 	
 	\item[c] Proveja condições suficientes para que o estimador de MQO sugerido seja consistente para o parâmetro de interesse, num regime assintótico em que $M \to \infty$. Qual a interpretação dessas condições?
 	
 	\item[d] Derive a variância assintótica do estimador, sob as hipóteses do item anterior, permitindo que haja correlação arbitrária entre os $\epsilon_{c,m}$ intramunicipais. Sugira um estimador consistente para a variância assintótica.
 \end{itemize}

 \paragraph{Exercício 9} Suponha que temos uma amostra $\{(y_{it}, x_{it})_{t=1}^T\}_{i=1}^N$. As observações $(y_{it},x_{it})_{t=1}^T$ são i.i.d. \emph{entre indivíduos ($i$)}, mas não impomos restrições sobre a relação entre $(y_{it}, x_{it})$ ao longo de $t$ para um mesmo $i$ (você pode pensar que selecionamos uma amostra aleatória de $N$ indivíduos, para os quais observamos $y$ e $x$ ao longo de $T$ períodos). Estamos interessados no modelo linear:
 
 \begin{equation}
 	y_{it} =  \beta_0 + \beta_1 x_{it} + \epsilon_{it} 
 	\quad \mathbb{E}[\epsilon_{it}]=0 
 	\quad \mathbb{E}[\epsilon_{it} x_{it}] = 0
 \end{equation}
 
 \begin{itemize}
 	\item[a] Suponha que $\mathbb{E}[\epsilon_{ik}\epsilon_{il}|x_{ik},x_{il}] = \sigma^2$ se $k = l$ e $\mathbb{E}[\epsilon_{ik}\epsilon_{il}|x_{ik},x_{il}] = 0 $ caso contrário. Mostre que, para $T$ fixo, quando $N \to \infty$, 
 	\[
 	\sqrt{N}(\hat{\beta}_1^\text{OLS} - \beta_1) \overset{d}{\to} N\left(0, V_1  \right),
 	\]
 	onde 
 	\[
 	V_1 \coloneqq \frac{\sigma^2 T^{-1}}{T^{-1}\sum_{t=1}^T\mathbb{E}[x_{it} - T^{-1}\sum_{t=1}^T \mathbb{E}[x_{it}]]^2}.
 	\]
 	O denominador de $V_1$ é conhecido como \emph{within-individual variance} de $X$, que denotamos por $\tau$.
 	
 	\item[b] Agora relaxe a hipótese anterior e suponha que $\mathbb{E}[\epsilon_{ik}\epsilon_{il}|x_{ik},x_{il}] = \sigma^2$ se $k = l$ e $\mathbb{E}[\epsilon_{ik}\epsilon_{il}|x_{ik},x_{il}] = \rho \sigma^2 $ se $k \neq l$. Mostre que, nesse caso, para $T$ fixo, quando $N \to \infty$,
 	\[
 	\sqrt{N}(\hat{\beta}_1^\text{OLS} - \beta_1) \overset{d}{\to} N(0, V_2),
 	\]
 	onde
 	\[
 	V_2 \coloneqq V_1 \left(1 + \rho \kappa (T-1)  \right)
 	\]
 	e 
 	\[
 	\kappa \coloneqq 
 	\frac{\sum_{k=1}^T \sum_{l\neq k }\mathbb{E}\left[\left(x_{ik} - T^{-1}\sum_{t=1}^T \mathbb{E}[x_{it}]\right)\left(x_{il} - T^{-1}\sum_{t=1}^T \mathbb{E}[x_{it}]\right)\right]}{T(T-1) \tau }
 	\]
 	é a correlação intra-indivíduo de $x$. Em quais casos $V_2$ é igual a $V_1$?
 	
 	\item[c] Mostre que, sob as hipóteses de (a), o estimador usual de variância computado sob homocedasticidade estima consistentemente $V_1$. Sob as hipóteses de (b), você consegue encontrar um estimador consistente para $V_2$? Mostre sua consistência.
 	
 	\item[d] Agora volte ao item (a) e relaxe a hipótese de homocedasticidade, mantendo ainda que $\mathbb{E}[\epsilon_{ik}\epsilon_{il}|x_{ik},x_{il}] = 0 $ se $k \neq l$. Qual é a variância assintótica ($V_3$) nesse caso?
 	
 	\item[e] Agora volte ao item (b) e relaxe a hipótese de homocedasticidade na estrutura de covariância. Qual é a variância assintótica ($V_4$) nesse caso? Em quais situações ela é igual à variância em (d)? Em geral, essas condições são as mesmas do item (b)?
 	
 	\item[f] Mostre que, sob as hipóteses do item (d), o estimador robusto usual de variância estima consistentemente a variância assintótica $V_3$. Sugira um estimador consistente para $V_4$ (sob as hipóteses do item (e)). Não é necessário provar sua consistência.
 \end{itemize}

 \paragraph{Exercício 11: Aproximação não parámetrica.}  Considere $Y$ uma variável aleatória real, e $X$ um vetor aleatório de dimensão $k$, ambos definidos num mesmo espaço de probabilidade $(\Omega, \Sigma, \mathbb{P})$. Seja $m$ a função de esperança condicional de $Y$ em $X$, i.e. $m(X) = \mathbb{E}[Y|X]$. Dizemos que uma coleção de funções $p_l: \mathbb{R}^k \mapsto \mathbb{R}$, $l \in \mathbb{N}$, aproxima $m$ em $L_2(\mathbb{P}_X)$ se existe uma sequência $\beta_j \in \mathbb{R}^j$, $j \in \mathbb{N}$, tal que:
 
 $$\lim_{j \to \infty}\int (m(x) - \beta_j'\boldsymbol{P}_j(x))^2 \mathbb{P}_X(dx) = 0$$
 com $\boldsymbol{P}_j(x) = (p_1(x),p_2(x),\ldots p_j(x))'$, e onde $\mathbb{P}_X$ denota a lei de probabilidade induzida por $X$.
 
 Seja $(X_i, Y_i)$, $i=1,\ldots, n$ uma amostra aleatória de $(X,Y)$. Dada uma coleção $\{p_l\}_{l\in \mathbb{N}}$ que aproxima $m$ em $L_2(\mathbb{P}_X)$, defina, para $J \in \mathbb{N}$, o estimador de MQO de $Y_i$ em $\boldsymbol{P}_J(X_i)$ como:
 
 $$\hat{b}_{J}  = \left(\boldsymbol{Z}_J' \boldsymbol{Z}_{J}\right)^{-1}\boldsymbol{Z}_J'\boldsymbol{y}\,, $$
 onde: $$\boldsymbol{y} = \begin{pmatrix}
 	Y_1\\
 	\vdots \\
 	Y_n
 \end{pmatrix}$$ e $$\boldsymbol{Z}_J = \begin{bmatrix}
 	\boldsymbol{P}_J(X_1) '\\
 	\boldsymbol{P}_J(X_2)' \\
 	\vdots \\
 	\boldsymbol{P}_J(X_n)'
 \end{bmatrix}\, .$$ No que segue, suporemos sempre que $J \leq n$, e que $\left(\boldsymbol{Z}_{J}' \boldsymbol{Z}_{J}\right)^{-1}$ existe com probabilidade 1. Além disso, como estaremos interessados nas propriedades do espaço gerado pelas colunas de $\boldsymbol{Z}_{J}$, vamos supor, sem perda de de generalidade, que $\frac{1}{n}\left(\boldsymbol{Z}_{J}'\boldsymbol{Z}_{J}\right) = \mathbb{I}_{J \times J}$.\footnote{De fato, o espaço gerado pelas colunas de $\boldsymbol{Z}_{J}$ é igual ao espaço gerado pelas colunas de $\tilde{\boldsymbol{Z}}_J = \sqrt{n}\boldsymbol{Z}_{J} (\boldsymbol{Z}_J'\boldsymbol{Z}_J)^{-1/2}$.}
 
 \begin{itemize}
 	\item[a] Seja $\beta_J \in \mathbb{R}^J$ o $J$-ésimo elemento da sequência de coeficientes que satisfaz a propriedade de aproximação de $m$. Mostre que:
 	
 	$$\lVert \hat{b}_{J} - \beta_J \rVert_2 \leq \lVert  (\boldsymbol{Z}_J'\boldsymbol{Z}_J)^{-1}\boldsymbol{Z}_J'\left(\boldsymbol{m} -\boldsymbol{Z}_J \beta_J\right)\rVert_2 + \lVert  (\boldsymbol{Z}_J'\boldsymbol{Z}_J)^{-1}\boldsymbol{Z}_J'\boldsymbol{\epsilon}\rVert_2$$ 
 	
 	onde $$\boldsymbol{m} = \begin{bmatrix}
 		m(X_1) \\
 		m(X_2) \\
 		\vdots \\
 		m(X_n)
 	\end{bmatrix}$$ e $$\boldsymbol{\epsilon} = \begin{bmatrix}
 		Y_1 - m(X_1)\\
 		Y_2 - m(X_2) \\
 		\vdots\\
 		Y_n - m(X_n)
 	\end{bmatrix}\, .$$ Mostre, além disso, que que:
 	\vspace{1em}
 	
 	$$\lVert  (\boldsymbol{Z}_J'\boldsymbol{Z}_J)^{-1}\boldsymbol{Z}_J'\left(\boldsymbol{m} -\boldsymbol{Z}_J \beta_J\right)\rVert_2^2  = \frac{1}{n}(\boldsymbol{m}-\boldsymbol{Z}_J\beta_J)'\boldsymbol{\Pi}_J(\boldsymbol{m}-\boldsymbol{Z}_J\beta_J)$$
 	
 	$$\lVert  (\boldsymbol{Z}_J'\boldsymbol{Z}_J)^{-1}\boldsymbol{Z}_J'\boldsymbol{\epsilon}\rVert_2^2  = \frac{1}{n}\boldsymbol{\epsilon}'\boldsymbol{\Pi}_J\boldsymbol{\epsilon}$$
 	
 	onde $\boldsymbol{\Pi}_J$ é a matriz de projeção de $\boldsymbol{Z}_J$.
 	\item[b]  Mostre que, se existe $C$ tal que $\mathbb{V}[Y|X] < C$ com probabilidade 1, então, com probabilidade 1:
 	
 	$$\mathbb{E}\left[\boldsymbol{\epsilon}'\boldsymbol{\Pi}_J \boldsymbol{\epsilon}|\boldsymbol{X}\right] \leq C J$$
 	onde $\boldsymbol{X} = [X_1 X_2 \ldots X_n]'$. \textit{Dica:} use  a decomposição espectral de uma matriz simétrica.
 	\item[c]  Mostre que:
 	$$\mathbb{E}[(\boldsymbol{m}-\boldsymbol{Z}_J\beta_J)'\boldsymbol{\Pi}_J(\boldsymbol{m}-\boldsymbol{Z}_J\beta_J)] \leq n \mathbb{E}[(m(X) - \beta_J'\boldsymbol{P}_J(X))^2]$$
 	\item[d]  Conclua que, sob o limite a $\mathbb{V}[Y|X]$ anterior, se considerarmos uma sequência $(J_n)_{n \in \mathbb{N}}$ tal que, quando $n \to \infty$, $J_n \to \infty$ e $\frac{J_n}{n} \to 0$:
 	
 	$$\lim_{n \to \infty}\mathbb{E}[\lVert \hat{b}_{J_n} - \beta_{J_n} \rVert_2^2] = 0 \, .$$
 	
 	Use a desigualdade de Markov para concluir que $\lVert \hat{b}_{J_n} - \beta_{J_n} \rVert_2 \overset{p}{\to} 0$.
 	
 	\item[e] Use os resultados anteriores para concluir que, sob as hipóteses em $\mathbb{V}[Y|X]$ e no crescimento de $J_n$:
 	
 	$$\lim_{n\to \infty}\mathbb{E}\left[\int(m(s) - \hat{b}_{J_n}'\boldsymbol{P}_{J_n}(s))^2\mathbb{P}_X(ds)\right] = 0\, ,$$
 	onde a expectativa de fora se dá com respeito à distribuição do estimador  $\hat{b}_{J_n}$ em amostras repetivas. Use a desigualdade de Markov para concluir que:
 	
 	$$\int(m(s) - \hat{b}_{J_n}'\boldsymbol{P}_{J_n}(s))^2\mathbb{P}_X(ds) \overset{p}{\to} 0 \, .$$
 	
 	Qual a interpretação desse resultado?
 \end{itemize}
